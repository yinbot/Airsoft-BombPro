00060 void LiquidCrystal_I2C_ByVac::setBacklight( uint8_t value ) 
00061 {
00062   Wire.beginTransmission(_Addr);
00063   Wire.write(0x03);                                     //  ByVac command code 0x03 for backlight
00064   if (value==0) Wire.write(1); else Wire.write((byte)0);        // 1 for off since polarity is NEGATIVE
